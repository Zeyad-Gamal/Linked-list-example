#include<iostream>
using namespace std;

struct Node{
	int data;
	Node* next;
};

    // Show List function................
 void show_list(Node* p){
 	Node* temp = p;
 	while(temp != NULL){
 		cout<<temp->data;
 		temp = temp->next;
	 }
 }
 
 // Search List function................
 
 void search_list(Node* p){
 	Node* temp = p;
 	int n;
 	cout<<"\nPlease Enter the value you want to search: ";
 	cin >> n;
 	int index , t=0 , f=0;
 	 while(temp != NULL){
 	 	if(temp -> data == n){
 	 		
			  f=1;
 	 		index =t;
 	 		
		  }
		  temp = temp -> next;
		  t++;
	  }
	  if(f==1){
	  	cout<<"\n"<<n<<" is the value and it's position is "<<index<<endl;
	  }
	  else{
	  	cout<<"\nThe value is not in the linked list"<<endl;
	  }
 }
